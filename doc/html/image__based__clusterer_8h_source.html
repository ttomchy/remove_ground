92  std::cout<<"Now it runs int the image_lable_cluster.h on new object received"<<std::endl;
93  // generate a projection from a point cloud
94  if (!cloud.projection_ptr()) {
95  fprintf(stderr, "ERROR: projection not initialized in cloud.\n");
96  fprintf(stderr, "INFO: cannot label this cloud.\n");
97  return;
98  }
99  time_utils::Timer timer;
100  std::cout<<"now it is runs before the LabelerT image_labeler(cloud.projection_ in the image_based_clustered.h !!! "<<std::endl;
101  LabelerT image_labeler(cloud.projection_ptr()->depth_image(),
102  cloud.projection_ptr()->params(), _angle_tollerance);
103  std::cout<<"now it is runs after the LabelerT image_labeler(cloud.projection_ in the image_based_clustered.h !!! "<<std::endl;
104  std::cout<<"now it is runs before the ComputeLabels function in the image_based_clustered.h !!! "<<std::endl;
105  image_labeler.ComputeLabels(_diff_type);
106  std::cout<<"now it is runs after the ComputeLabels function in the image_based_clustered.h !!! "<<std::endl;
107  std::cout<<"now it is runs before the image_labeler.GetLabelImage() function in the image_based_clustered.h !!! " <<std::endl;
108  const cv::Mat* labels_ptr = image_labeler.GetLabelImage();
109  std::cout<<"now it is runs after the image_labeler.GetLabelImage() function in the image_based_clustered.h !!! " <<std::endl;
110  fprintf(stderr, "INFO: image based labeling took: %lu us\n",
111  timer.measure());
112 
113  // send image to whoever wants to get it
114  if (_label_client) {
115  _label_client->OnNewObjectReceived(*labels_ptr, this->id());
116  }
117 
118  // create 3d clusters from image labels
119  std::unordered_map<uint16_t, Cloud> clusters;
120  for (int row = 0; row < labels_ptr->rows; ++row) {
121  for (int col = 0; col < labels_ptr->cols; ++col) {
122  const auto& point_container = cloud.projection_ptr()->at(row, col);
123  if (point_container.IsEmpty()) {
124  // this is ok, just continue, nothing interesting here, no points.
125  continue;
126  }
127  uint16_t label = labels_ptr->at<uint16_t>(row, col);
128  if (label < 1) {
129  // this is a default label, skip
130  continue;
131  }
132  for (const auto& point_idx : point_container.points()) {
133  const auto& point = cloud.points()[point_idx];
134  clusters[label].push_back(point);
135  }
136  }
137  }
138 
139  // now send clusters to clients. They can't wait to get a new cloud.
140  //现在开始发送给新的客户端， they cann't wait to get the point .
141  auto labels_color = AbstractImageLabeler::LabelsToColor(*labels_ptr);
142  _counter++;
143  std::vector<Cloud> clusters_to_send;
144  for (auto& kv : clusters) {
145  // if (counter > num_clusters_max) { break; }
146  auto& cluster = kv.second;
147  if (cluster.size() < this->_min_cluster_size ||
148  cluster.size() > this->_max_cluster_size) {
149  // cluster is too small or too big, forget it.
150  continue;
151  }
152  cluster.InitProjection(cloud.projection_ptr()->params());
153  clusters_to_send.push_back(cluster);
154  }
155  std::cout<<"Now it runs in last of the image_based_clustered.cpp !!!"<<std::endl;
156  this->ShareDataWithAllClients(clusters_to_send);
157  }
